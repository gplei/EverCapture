CREATE TABLE transactions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    date DATE,                       -- 交易日期
    description VARCHAR(255),        -- 交易说明
    amount DECIMAL(10,2),           -- 金额：expense/income为正数，refund为负数
    type ENUM('expense', 'income', 'transfer') NOT NULL,  -- 类型
    account_from VARCHAR(50),                 -- 来源账户，例如 CreditCard / BankA / Cash / Friend
    account_to VARCHAR(50),                   -- 目标账户，例如 Bank / CreditCard / Cash / Friend
    category VARCHAR(50),                     -- 分类：Grocery / Travel / Salary / Transfer等
    source VARCHAR(50),                       -- 交易来源/账单，例如 'BankA Statement', 'AppleCard Statement', 'Cash'
    owner varchar(20),
    note TEXT,                                 -- 备注，可用于标记退款或特殊情况
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP

);

// select txn and txn_detail with no detail lines
select t.id, t.date, ifnull(d.item, t.description), ifnull(c.opt_name, tc.opt_name),
t.description, d.item, t.category_id, tc.opt_name, d.category_id, c.opt_name 
from transaction_log t 
left join transaction_detail_log d on d.transaction_id=t.id
left join transaction_option c on c.type='Category' and c.id=d.category_id 
left join transaction_option tc on tc.type='Category' and tc.id=t.category_id 
where t.id in (1,2,3);

// create category table
drop table txn_categories;
CREATE TABLE txn_categories (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(100) NOT NULL,         -- category or subcategory name
  parent_id INT NULL,                  -- NULL = top-level category
  description VARCHAR(255) DEFAULT NULL,
  active BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
